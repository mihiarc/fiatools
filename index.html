Quick start

Get running in minutes with pip install and a few lines of code.

# Install the package
pip install pyfia

# Then use it
from pyfia import FIA, biomass, tpa, volume

# Connect to FIA database
with FIA("path/to/FIA_database.duckdb") as db:
    # Filter to North Carolina
    db.clip_by_state(37)
    db.clip_most_recent(eval_type="EXPVOL")

    # Get forest metrics
    tpa_results = tpa(db, tree_domain="STATUSCD == 1")
    biomass_results = biomass(db, by_species=True)
    volume_results = volume(db, land_type="forest")
# Install the package
pip install gridfia

# Then use it
from gridfia import GridFIA

# Initialize with a state
gf = GridFIA(state="NC")

# Get species presence data
species_data = gf.get_species_presence("Pinus taeda")

# Calculate diversity indices
diversity = gf.calculate_diversity(
    metrics=["shannon", "simpson"]
)
# Install the package
pip install pyfvs-fia

# Then use it
from pyfvs import Stand, Tree

# Create a stand with site index
stand = Stand(site_index=70, site_species="loblolly")

# Add trees from inventory
stand.add_trees([
    Tree(species="loblolly", dbh=8.5, height=55),
    Tree(species="loblolly", dbh=10.2, height=62),
])

# Project growth for 20 years
stand.simulate(years=20, cycle_length=5)
yield_table = stand.get_yield_table()
